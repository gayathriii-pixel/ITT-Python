def verify_password(password):
    if len(password) < 8:
        return "PASSWORD CONTAINS ATLEAST 8 CHARACTERS"

    has_letter = False
    has_digit = False
    has_special = False

    for ch in password:
        if ch.isalpha():
            has_letter = True
        elif ch.isdigit():
            has_digit = True
        else:
            has_special = True

    if has_letter and has_digit and has_special:
        return "VALID PASSWORD"
    else:
        return "Password must contain letters, numbers, and special characters"


# Input from user
pwd = input("Enter the password: ")
print(verify_password(pwd))
